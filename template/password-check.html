JavaScript密码验证演示

体验浏览器原生弹窗的密码验证功能

这个页面演示了使用JavaScript的prompt()和alert()函数实现的简单密码验证功能。

点击下面的按钮将触发密码验证流程，您将看到浏览器原生的弹窗对话框。

演示区域

点击按钮开始密码验证：

提示：正确密码是 "myz"

功能说明

  • 点击按钮后，将弹出密码输入对话框
  • 您有3次尝试机会
  • 输入正确密码将显示成功消息
  • 3次错误输入后将自动返回上一页
  • 点击"取消"按钮也会返回上一页

JavaScript代码

function password() {
    var testV = 0;
    var pass1 = prompt('赶紧输密码:', '');
    
    while (testV < 3) {
        if (!pass1) history.go(-1);
        
        if (pass1 == "myz") {
            alert('密码正确!');
            break;
        }
        
        testV += 1;
        
        if (testV == 1) {
            pass1 = prompt('密码错了，搞什么啊！还剩两次机会。');
        } else if (testV == 2) {
            pass1 = prompt('密码错了，搞什么啊！还剩一次机会。');
        }
    }
    
    if (pass1 != "myz" && testV == 3) {
        history.go(-1);
    }
    
    return " ";
}

// 绑定按钮点击事件
document.getElementById('passwordBtn').addEventListener('click', password);
